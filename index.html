// ARRAY_OPS.CPP 
#include <iostream>
using namespace std;

void insertElement(int arr[], int& n, int x, int pos) {
    for (int i = n - 1; i >= pos; i--) {
        arr[i + 1] = arr[i];
    }
    arr[pos] = x;
    n++;
}

int main() {
    int data[10] = {10, 20, 30, 40};
    int size = 4;
    
    // Injecting new logic...
    insertElement(data, size, 25, 2);
    
    cout << "> SYSTEM READY." << endl;
    return 0;
}
